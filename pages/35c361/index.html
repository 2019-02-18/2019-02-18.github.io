# 1. 初始化蓝牙

使用 uni.openBluetoothAdapter(OBJECT) 方法初始化蓝牙。

uni.openBluetoothAdapter({
  success(res) {
    console.log("初始化蓝牙成功");
  },
  fail(err) {
    console.log("初始化蓝牙失败", err);
  },
});

# 2. 搜索蓝牙设备

使用 uni.startBluetoothDevicesDiscovery(OBJECT) 方法开始搜索蓝牙设备。 安卓如果搜索不到蓝牙设备，需要开启手机位置信息。 此操作比较耗费系统资源，请在搜索并连接到设备后调用 uni.stopBluetoothDevicesDiscovery(OBJECT) 方法停止搜索。

uni.startBluetoothDevicesDiscovery({
  services: [], // 要搜索的服务列表
  allowDuplicatesKey: false, // 是否允许重复上报同一设备,但RSSI值会有所不同
  success(res) {
    console.log("开始搜索蓝牙设备成功", res);
  },
  fail(err) {
    console.log("开始搜索蓝牙设备失败", err);
  },
});

# 3. 发现外围设备

使用 uni.onBluetoothDeviceFound(CALLBACK) 方法监听发现的外围设备。

uni.onBluetoothDeviceFound(function (res) {
  res.devices.forEach((device) => {
    //过滤掉没有名字的设备
    if (!device.name && !device.localName) {
      return;
    }
    console.log("发现外围设备", res);
    // }
  });
});

# 6. 获取蓝牙设备所有服务

使用 uni.getBLEDeviceServices(OBJECT) 方法获取蓝牙设备所有服务。 在调用 getBLEDeviceServices 的时不要直接调用一定要设置时间间隔，延迟调用

setTimeout(() => {
  uni.getBLEDeviceServices({
    deviceId: "", // 要连接的设备 id
    success(res) {
      console.log("获取蓝牙设备所有服务成功", res);
    },
    fail(err) {
      console.log("获取蓝牙设备所有服务失败", err);
    },
  });
}, 2000);

# 8. 启用蓝牙设备特征值变化时的 notify 功能

使用 uni.notifyBLECharacteristicValueChange(OBJECT) 方法启用蓝牙设备特征值变化时的 notify 功能。必须设备的特征值支持 notify 或者 indicate 才可以成功调用。APP 调用 uni.createBLEConnection 之后需要延时 2 秒再调用 uni.notifyBLECharacteristicValueChange。

uni.notifyBLECharacteristicValueChange({
  deviceId: "", // 要连接的设备 id
  serviceId: "", // 要操作的服务 uuid
  characteristicId: "", // 要操作的特征 uuid
  state: true, // 启用或停用 notify
  success(res) {
    console.log("启用 notify 成功", res);
  },
  fail(err) {
    console.log("启用 notify 失败", err);
  },
});

# 9. 监听低功耗蓝牙设备的特征值变化

CALLBACK 返回参数

属性 类型 说明
deviceId string 蓝牙设备 id
serviceId string 蓝牙特征值对应服务的 uuid
characteristicId string 蓝牙特征值的 uuid
value ArrayBuffer 特征值最新的值

使用 uni.onBLECharacteristicValueChange(CALLBACK) 方法监听低功耗蓝牙设备的特征值变化。必须先启用 notifyBLECharacteristicValueChange 接口才能接收到设备推送的 notification

# 重点！！！！

let receivedData = '';//接收到的数据
uni.onBLECharacteristicValueChange(function (res) {
    try {
        console.log("特征值变化", res);
        //注意：这里的res.value是ArrayBuffer类型，需要进行数据转换下面提供两次方法根据需求转换
        let resStr = ab2str(res.value);
        receivedData += resStr;//注意蓝牙返回的数据可能会分包，所以需要拼接
        if (receivedData.endsWith('}')) {//判断数据是否接收完成我这里是以转换后以}结尾
            //数据已经接收完成，根据需求进行数据处理
            let jsonData = receivedData.;
            console.log('接收到的完整数据',jsonData)
            // 重置receivedData变量
            receivedData = '';
        }
    } catch (e) {
        console.log(e)
    }
});
// ArrayBuffer转16进度字符串示例
function ab2hex(buffer) {
  const hexArr = Array.prototype.map.call(
    new Uint8Array(buffer),
    function (bit) {
      return ("00" + bit.toString(16)).slice(-2);
    }
  );
  return hexArr.join("");
}
// ArrayBuffer直接转字符串示例
function ab2str(buffer) {
    return String.fromCharCode.apply(null, new Uint8Array(buf));
},

# 10. 对需要操作的特征值进行读、写操作

使用 uni.readBLECharacteristicValue(OBJECT) 方法读取特征值，使用 uni.writeBLECharacteristicValue(OBJECT) 方法写入特征值。

// 读取特征值
uni.readBLECharacteristicValue({
  deviceId: "", // 要连接的设备 id
  serviceId: "", // 要操作的服务 uuid
  characteristicId: "", // 要操作的特征 uuid
  success(res) {
    console.log("读取特征值成功", res);
  },
  fail(err) {
    console.log("读取特征值失败", err);
  },
});

// 写入特征值
uni.writeBLECharacteristicValue({
  deviceId: "", // 要连接的设备 id
  serviceId: "", // 要操作的服务 uuid
  characteristicId: "", // 要操作的特征 uuid
  value: new ArrayBuffer(0), // 要写入的数据
  success(res) {
    console.log("写入特征值成功", res);
  },
  fail(err) {
    console.log("写入特征值失败", err);
  },
});
